\section{Boltzmann's Distribution}
\subsection{Gas Example}
$N$ Gas particles occupy small volumetric cells $i=1,2,3,\cdots,s$ of phase space, with occupation number $n_i$. Particle number is conserved so
\begin{equation}
  \sum_i n_i = N
\end{equation}
\begin{figure}
\centering
\includegraphics[scale=0.5]{./figures/ch1/gas_particles.pdf}
\end{figure}
The number of ways $W$ that any particular distribution $\{n_1, n_2, ..., n_s \}$ of particles will fall within their respective volumetric cells in phase space is given by the multinomial formula
\begin{equation}
  W = \frac{N!}{(n_1!n_2!...n_s!)}
\end{equation}
Taking the logarithm of $W$ and approximating the factorial for large $N$ using Stirlingâ€™s formula gives
\begin{equation}
  \log{W} = \log{N!}-\sum_i \log{n_i!} \approx -N\sum_i \frac{n_i}{N}\log{(\frac{n_i}{N})} = -N\sum_i p_i \log{p_i}
\end{equation}
where $p_i=\frac{n_i}{N}$ is taken to be the probability that a particle is in cell $i$, provided $N$ is sufficiently large. The entropy $S$ is defined by
\begin{equation}
  S = -k_B \sum_i p_i \log{p_i} 
\end{equation}
There are two constraints
\begin{align}
  \sum_i p_i &= 1\\
  \sum_i p_i \epsilon_i &= \bar{\epsilon}
\end{align}
where $\epsilon_i$ is the total energy of cell $i$, and $\bar{\epsilon}$ is the average energy per particle.

\section{Shannon's information}
Consider a signal that passes through a commnication channel
\begin{figure}[h]
  \centering
  \includegraphics[scale=0.6]{./figures/ch1/shannon_channel.pdf}
  \caption{Shannon's channel}
\end{figure}
Suppose the message is a linear string of symbols that is $M$ characters long, where each character is drawn independently from an $r$-letter alphabet, with probability $p_i$. Let $m_i(i=1,2,3,...,r)$ represent the number of times that the ith type of character is observed in the message, as shown in Figure~\ref{fig:shannonconcept}.
\begin{figure}[h!]
  \centering
  \includegraphics[scale=0.6]{./figures/ch1/shannon_concept_illu.pdf}
  \caption{The illustration of message}
  \label{fig:shannonconcept}
\end{figure}

When $M$ is large, the most likely $M$-letter message will have the composition $m_i=Mp_i$, and this occurs with probability
\begin{equation}
  P \equiv p_1^{m_1}\cdots p_{r}^{m_r}=p_1{Mp_1}\cdots p_r{Mp_r}
\end{equation}
\begin{figure}[h!]
  \centering
  \includegraphics[scale=0.55]{./figures/ch1/MBigcase.pdf}
  \caption{When $M$ is large}
  \label{fig:mbigcase}
\end{figure}

In the limiting case of an alphabet having only a single letter, the probability of knwoing the message is 
\begin{equation}
  P=1
\end{equation}
because there is only one possible string of characters.
\begin{figure}
  \centering
  \includegraphics[scale=0.55]{./figures/ch1/MSmallcase.pdf}
  \caption{When the alphabet pool only has one letter.}
  \label{fig:msmallcase}
\end{figure}
The larger the alphabet is($r$ becomes larger), the smaller the value of $P$ and the greater the uncertainty of receiving a particular message. The logarithm of $1/P$
\begin{equation}
  H=\log{(1/P)}=-M\sum_i p_i \log{p_i}
\end{equation}
is sometimes called the uncertainty or missing information.
\begin{figure}
  \centering
  \includegraphics[scale=0.55]{./figures/ch1/alphabetlarge.pdf}
  \caption{When the alphabet pool becomes larger, the larger the uncertainty}
  \label{fig:alphabetlarge}
\end{figure}
